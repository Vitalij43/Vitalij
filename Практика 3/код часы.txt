#include <8051.h>
#include <string.h>
#include <time.h>

void tick(short unsigned int* time) {
	time[5] += 1;	
	if(time[5] >= 10){
		time[5] = 0;
		time[4] += 1;
	}
	if(time[4] >= 6){
		time[4] = 0;
		time[3] += 1;
	}
	if (time[3] >= 10) {
		time[3] = 0;
		time[2] += 1;
	}
	if (time[2] >= 6) {
		time[2] = 0;
		time[1] += 1;
	}
	if (time[1] >= 10) {
		time[1] = 0;
		time[0] += 1;
	}
	if ((time[0] == 2) && (time[1] == 4)) {
		time[0] = 0;
		time[1] = 0;
	}
}
